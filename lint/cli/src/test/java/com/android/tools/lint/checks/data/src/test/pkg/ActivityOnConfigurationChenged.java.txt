package test.pkg;

import android.app.Activity;
import android.content.res.Configuration;

public class ActivityOnConfigurationChenged {
    private static class Test1 extends Activity {
        protected void onConfigurationChanged(Configuration configuration) {
            // Error
        }
    }

    private static class Test2 extends Activity {
        protected void onConfigurationChanged(int foo) {
            // OK: not overriding the right method
        }
    }

    private static class Test3 extends Activity {
        protected void onConfigurationChanged(Configuration configuration) {
            // OK: Calling super
            super.onConfigurationChanged(configuration);
        }
    }

    private static class Test4 extends Activity {
        protected void onConfigurationChanged(Configuration configuration) {
            // Error: missing detach call
            int x = 1;
            x++;
            System.out.println(x);
        }
    }

}
